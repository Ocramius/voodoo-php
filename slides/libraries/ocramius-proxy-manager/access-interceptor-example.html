class Foo {
    public function doFoo()
    {
        echo "Foo!";
    }
}


$factory = new AccessInterceptorScopeLocalizerFactory();

$proxy = $factory->createProxy(
    new Foo(),
    ['doFoo' => function ($proxy) { echo "pre\n"; }],
    ['doFoo' => function ($proxy) { echo "post\n"; }]
);

$proxy->doFoo();
